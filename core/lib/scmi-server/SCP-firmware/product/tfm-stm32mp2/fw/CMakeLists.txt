# Create the firmware target.
#

add_library(${SCP_FIRMWARE_TARGET})

# cmake-lint: disable=E1122

target_include_directories(
    ${SCP_FIRMWARE_TARGET}
    PUBLIC  ${CMAKE_CURRENT_SOURCE_DIR}/../include
            ${SCP_TFM_PLAT_DIR}/native_driver/include
            ${SCP_TFM_PLAT_DIR}/../devicetree/include
            ${SCP_TFM_PLAT_DIR}/native_driver
            ${CMAKE_BINARY_DIR}/generated/devicetree/platform_s
            ${STM_SOC_DIR}/secure_fw/partitions/scp
)

target_sources(
    ${SCP_FIRMWARE_TARGET}
    PRIVATE "${CMAKE_CURRENT_SOURCE_DIR}/config_all.c"
)
